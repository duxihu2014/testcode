实现逻辑：参考下面的sql计算出明细，再通过last_modified_date计算当前搜索年的月份的统计值
select sd.last_modified_date ,ship.brand_code ,sd.ppid,  ore.PART_USE_STATUS,  ore.DOA_REASON, sd.*,ore.* from 
shipment_detail sd  
inner join ods_rejects_extend ore on sd.source_id = ore.REJECTS_ID 
inner join shipment ship on ship.id = sd.shipment_id  
where ore.PART_USE_STATUS='DOA'
