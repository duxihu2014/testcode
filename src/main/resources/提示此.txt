    @Override
    public ResponseEntity<?> downloadExcelTemplate() throws IOException {
        String templateFileName = "shiporder_template";
        String filePath = "excel-template/" + templateFileName + ".xlsx";
        ClassPathResource classPathResource = new ClassPathResource(filePath);
        InputStream inputStream = classPathResource.getInputStream();
        InputStreamResource resource =
                new InputStreamResource(inputStream);
        HttpHeaders headers = new HttpHeaders();
        headers.add("Cache-Control", "no-cache,no-store,must-revalidate");
        headers.add("Pragma", "no-cache");
        headers.add("Expires", "0");
        headers.add(HttpHeaders.CONTENT_DISPOSITION, "attachment; filename="+templateFileName+".xlsx");

        return ResponseEntity.ok()
                .headers(headers)
                .contentType(MediaType.APPLICATION_OCTET_STREAM)
                .<Object>body(resource);
    }
