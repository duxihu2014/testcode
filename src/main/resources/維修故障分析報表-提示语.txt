通过下面的sql获得尺寸，機種，工程師確認故障描述，維修及更換部件，这个sql不做统计计算

select pm.PARTS_LENGTH ,pm.SKU_NO ,rdr.ROOT_CAUSE,pm2.PRODUCT_TYPE  from [gssd-gsp_mes].dbo.repair_detail_record rdr
left join [gssd-gsp_mes].dbo.repair_detail rd   on rd.ID =rdr.DETAIL_ID 
LEFT JOIN (
    SELECT *, 
           ROW_NUMBER() OVER (PARTITION BY PART_NO ORDER BY (SELECT NULL)) as rn
    FROM [gssd-gsp_official_web].dbo.prod_master
) pm2
on rdr.ORG_PN =   pm2.PART_NO  AND pm2.rn = 1
LEFT JOIN (
    SELECT *, 
           ROW_NUMBER() OVER (PARTITION BY PART_NO ORDER BY (SELECT NULL)) as rn
    FROM [gssd-gsp_official_web].dbo.prod_master
) pm 
on rd.PART_NO =   pm.PART_NO AND pm.rn = 1
where pm.SKU_NO is not null and rdr.ORG_PN is not null and pm2.PRODUCT_TYPE is not null
and pm.PARTS_LENGTH =21
--order by pm.PARTS_LENGTH
;
